26 // Converts expressions from v1.0 to v1.2
27 // However, the type in each expression is still a v1.0 type.
28 class ExpressionConverter : public Transform {
29  protected:
30  ProgramStructure* structure;
31  P4::P4CoreLibrary &p4lib;
32  public:
33  bool replaceNextWithLast; // if true p[next] becomes p.last
34  explicit ExpressionConverter(ProgramStructure* structure)
35  : structure(structure), p4lib(P4::P4CoreLibrary::instance),
36  replaceNextWithLast(false) { setName("ExpressionConverter"); }
37  const IR::Type* getFieldType(const IR::Type_StructLike* ht, cstring fieldName);
38  const IR::Node* postorder(IR::Constant* expression) override;
39  const IR::Node* postorder(IR::Member* field) override;
40  const IR::Node* postorder(IR::FieldList* fl) override;
41  const IR::Node* postorder(IR::Mask* expression) override;
42  const IR::Node* postorder(IR::ActionArg* arg) override;
43  const IR::Node* postorder(IR::Primitive* primitive) override;
44  const IR::Node* postorder(IR::PathExpression* ref) override;
45  const IR::Node* postorder(IR::ConcreteHeaderRef* nhr) override;
46  const IR::Node* postorder(IR::HeaderStackItemRef* ref) override;
47  const IR::Node* postorder(IR::GlobalRef *gr) override;
48  const IR::Expression* convert(const IR::Node* node) {
49  auto result = node->apply(*this);
50  return result->to<IR::Expression>();
51  }
52 };

54 class StatementConverter : public ExpressionConverter {
55  std::map<cstring, cstring> *renameMap;
56  public:
57  StatementConverter(ProgramStructure* structure, std::map<cstring, cstring> *renameMap)
58  : ExpressionConverter(structure), renameMap(renameMap) {}
59 
60  const IR::Node* preorder(IR::Apply* apply) override;
61  const IR::Node* preorder(IR::Primitive* primitive) override;
62  const IR::Node* preorder(IR::If* cond) override;
63  const IR::Statement* convert(const IR::Vector<IR::Expression>* toConvert);
64 
65  const IR::Statement* convert(const IR::Node* node) {
66  auto conv = node->apply(*this);
67  auto result = conv->to<IR::Statement>();
68  BUG_CHECK(result != nullptr, "Conversion of %1% did not produce a statement", node);
69  return result;
70  }
